SELECT pd.rnames AS RATING_NAME,

  dnra.EXTERNAL_ID_1,

  rvalues.vcodes AS RATING_VALUES,

  re.REGION_CODE

FROM DN_RATING_ATTRIBUTE dnra

LEFT JOIN dn_property dnp

ON TO_CHAR(dnra.DN_RATING_ATTRIBUTE_ID) = SUBSTR(dnp.COLUMN_NAME, 4)

LEFT JOIN

  (SELECT dpd.DN_PROPERTY_ID,

    regexp_replace(LISTAGG(dpd.name

    || ' ('

    || dl.description

    || ')', ', ') WITHIN GROUP (

  ORDER BY dpd.DN_PROPERTY_ID),'(^|,)(.+)(,\2)+', '\1\2') AS rnames

  FROM dn_property_description dpd

  INNER JOIN dn_language dl

  ON dl.DN_LANGUAGE_ID = dpd.DN_LANGUAGE_ID

  GROUP BY dpd.DN_PROPERTY_ID

  ) pd ON dnp.DN_PROPERTY_ID = pd.DN_PROPERTY_ID

JOIN dn_all_code control

ON control.code = dnra.control_type

LEFT JOIN

  (SELECT dnra1.DN_RATING_ATTRIBUTE_ID,

    utl_i18n.unescape_reference(RTRIM(XMLAGG(XMLELEMENT(E,ratings.Rating_CODE

    || ' - '

    ||ratings.Rating_Value,', ').EXTRACT('//text()')

  ORDER BY dnra1.DN_RATING_ATTRIBUTE_ID).GetClobVal(),', ')) AS vcodes

  FROM dn_rating_attribute dnra1

  INNER JOIN

    ( SELECT DISTINCT dnra.DN_RATING_ATTRIBUTE_ID,

      v.dn_code_value_id,

      v.code        AS Rating_CODE,

      d.description AS Rating_Value

    FROM dn_rating_attribute dnra

    INNER JOIN dn_property dnp

    ON (SUBSTR(dnp.column_name,4)   = SUBSTR(dnra.dn_rating_attribute_id,1)

    AND SUBSTR(dnp.column_name,1,3) = 'rt_'

    AND dnp.tenant_id               = dnra.tenant_id)

    INNER JOIN dn_code_table t

    ON (t.code_role = dnp.column_name

    AND t.tenant_id = dnra.tenant_id)

    INNER JOIN dn_code_value v

    ON (v.dn_code_table_id = t.dn_code_table_id

    AND v.tenant_id        = dnra.tenant_id)

    INNER JOIN dn_code_description d

    ON (d.dn_code_value_id                    = v.dn_code_value_id

    AND d.tenant_id                           = dnra.tenant_id)

    WHERE dnp.table_name                      = 'vt_rating'

    AND dnra.tenant_id                        = 500000

    ) ratings ON dnra1.DN_RATING_ATTRIBUTE_ID = ratings.DN_RATING_ATTRIBUTE_ID

  GROUP BY dnra1.DN_RATING_ATTRIBUTE_ID

  ) rvalues ON dnra.dn_rating_attribute_id = rvalues.DN_RATING_ATTRIBUTE_ID

INNER JOIN region re

ON re.REGION_ID         = dnra.REGION_ID

WHERE dnra.STATUS       = 'ACTV'

AND dnp.table_name      = 'vt_rating'

AND dn_property_role_id = 9999000000002435

AND control.table_name  = 'dn_rating_attribute'

AND control.column_name = 'control_type'

AND control.language    = 'System'

ds.description script_description,
ds.script_type,
ds.script_platform,
ds.status,
dut.DESCRIPTION user_type_description,
dv.DESCRIPTION version_description
from dn_script ds
join DN_ENTITY_TABLE_SCRIPT dets on ds.script_id=dets.script_id
join dn_entity de on dets.dn_entity_id=de.dn_entity_id
join dn_user_type dut on dut.DN_USER_TYPE_ID=ds.DN_USER_TYPE_ID
join dn_version dv on dv.dn_version_id=ds.dn_version_id
where 1=1
and lower(ds.script_text) like '%meeting_attend%'
and lower(ds.script_text) like '%status%'
--and lower(ds.script_text) like '%attd%'
and ds.status='ACTV';
